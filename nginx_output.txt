Nginx config files: ['sovereign']

=== /etc/nginx/sites-enabled/sovereign ===
server {
    listen 80;
    server_name fectok.com vazo.fectok.com;

    # Core API Gateway
    location / {
        proxy_pass http://127.0.0.1:5000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection upgrade;
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    # Sovereign Media Tunnel [A_124]
    # Connects Mesh Users directly to the Global Video Hub
    location /video_stream/ {
        # Proxy to Uplink Hub (Docker internal or localhost 8080)
        proxy_pass http://127.0.0.1:8080/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        
        # Performance Tuning for 4K/HD Streaming
        proxy_buffering off;
        proxy_request_buffering off;
        client_max_body_size 500M;
    }
}
